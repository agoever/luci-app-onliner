<%# Copyright 2008 Steven Barth Copyright 2008-2011 Jo-Philipp Wich Licensed to the public under the Apache License 2.0. -%> <% local util = require "luci.util" local function online_data() local rv = { } local function normalize_mac(mac) return mac:gsub("[%-:]", ""):upper():gsub("(%w%w)", "%1:"):sub(1,17) end -- 获取桥接端口到接口名的映射（端口号从1开始） local port_to_iface = {} do local brif_dir = "/sys/class/net/br-lan/brif/" local dir = io.popen("ls " .. brif_dir .. " 2>/dev/null | sort") if dir then local port = 1 for iface in dir:lines() do port_to_iface[port] = iface port = port + 1 end dir:close() end end -- 预判接口频段（根据常见MTK命名） local iface_to_band = {} for port, iface in pairs(port_to_iface) do if iface:match("^ra%d") then iface_to_band[iface] = "2.4G" elseif iface:match("^rax") then iface_to_band[iface] = "5G" end end -- 获取MAC到端口号的映射（通过brctl） local mac_to_port = {} do local f = io.popen("brctl showmacs br-lan 2>/dev/null") if f then for line in f:lines() do local port, mac, local_flag = line:match("^%s*(%d+)%s+([:%x]+)%s+(%w+)") if port and mac and local_flag == "no" then mac_to_port[normalize_mac(mac)] = tonumber(port) end end f:close() end end -- 从 DHCP 租约文件中估算每个 MAC 的租约开始时间（假设默认租约12小时） local mac_to_start = {} do local lease_fd = io.open("/tmp/dhcp.leases", "r") if lease_fd then local default_lease = 43200 -- 12小时，单位秒 for line in lease_fd:lines() do local expiry, mac, ip, hostname = line:match("^(%d+) (%S+) (%S+) (%S+)") if expiry and mac then local expiry_time = tonumber(expiry) if expiry_time then local start_time = expiry_time - default_lease if start_time > 0 then mac_to_start[normalize_mac(mac)] = start_time end end end end lease_fd:close() end end -- 获取设备首次在线时间戳（通过临时文件缓存） local first_seen = {} local cache_file = "/tmp/luci_onliner_cache" local now = os.time() -- 读取缓存文件 local cache_fd = io.open(cache_file, "r") if cache_fd then for line in cache_fd:lines() do local mac, ts = line:match("^(%S+)%s+(%d+)$") if mac and ts then first_seen[mac] = tonumber(ts) end end cache_fd:close() end -- 获取在线设备数据 local fd = util.execi('/usr/bin/awk \'BEGIN{while ((getline < "/tmp/dhcp.leases") > 0){a[$2]=$4;}while ((getline < "/proc/net/arp") > 0){if (!a[$4]){a[$4]="\?";}if (match($3,"0x[26]")){"ping -q -c 1 "$1" &";if (b[$4]){b[$4]=b[$4]"/"$1;}else{b[$4]=$1;}c[$4]=$6;}}while (("ip -6 neighbor show | grep -v fe80" | getline) > 0){if (b[$5]) {"ping -q -c 1 "$1" &";b[$5]=b[$5]"/"$1;}}for (mac in b){print(a[mac],b[mac],mac,c[mac]);}}\' ') local current_macs = {} while true do local ln = fd() if ln == nil then break end local name,ip,mac,dev = ln:match("^(%S+) (%S+) (%S+) (%S+)") if mac and ip and name and dev then local conn_type = "有线" local mac_upper = normalize_mac(mac) local reason = "默认有线" local port = mac_to_port[mac_upper] -- 无线判断：根据端口号区分2.4G/5G if port == 5 or port == 6 then local iface = port_to_iface[port] if iface and iface_to_band[iface] then conn_type = iface_to_band[iface] -- "2.4G" 或 "5G" reason = "brctl " .. iface .. " " .. conn_type else conn_type = "无线" reason = "brctl 无线通道(port " .. port .. ")" end end -- 记录当前在线的MAC current_macs[mac_upper] = true -- 计算在线时长（分钟） local online_minutes = 0 if first_seen[mac_upper] then online_minutes = math.floor((now - first_seen[mac_upper]) / 60) else -- 尝试从 DHCP 估算时间中获取 local dhcp_start = mac_to_start[mac_upper] if dhcp_start and dhcp_start > 0 and dhcp_start <= now then first_seen[mac_upper] = dhcp_start online_minutes = math.floor((now - dhcp_start) / 60) else first_seen[mac_upper] = now online_minutes = 0 end end rv[#rv+1] = { hostname = name, device = dev, macaddr = mac, ipaddr = ip, conn_type = conn_type, debug_reason = reason, online_minutes = online_minutes } end end -- 清理缓存：移除已不在线的设备，保留在线的设备时间戳 local new_cache = {} for mac, ts in pairs(first_seen) do if current_macs[mac] then new_cache[mac] = ts end end -- 写回缓存文件 local cache_fd_w = io.open(cache_file, "w") if cache_fd_w then for mac, ts in pairs(new_cache) do cache_fd_w:write(mac .. " " .. ts .. "\n") end cache_fd_w:close() end return { onlines = rv, probe = { hostapd_cli=0, ubus=0, iwinfo=0 } } end if luci.http.formvalue("status") == "1" then local data = online_data() luci.http.prepare_content("application/json") luci.http.write_json(data) return end %> <%+header%>
